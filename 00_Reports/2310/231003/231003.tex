$\blacksquare$\;\gt{粒子位置の計算}
\begin{eqnarray*}
  y'_i = \frac{1}{N_i} \sum_{j=1}^{N} y_{ij} &:& y方向位置  \\
  z'_i = \frac{1}{N_i} \sum_{j=1}^{N} z_{ij} &:& z方向位置  \\
  n'_i = \frac{1}{N_i} \sum_{j=1}^{N} n_{ij} &:& フレーム数 \\
  i &:& \text{クラスタ番号}\\
  j &:& \text{フレーム番号}\\
  N_i &:& i
\end{eqnarray*}

$\blacksquare$\;\gt{主流方向速度の推定}
\begin{eqnarray*}
  u_i &=& T \times \frac{f}{N_i}\\
  T &:& \text{レーザーシート厚み} \\
  f &:& \text{フレームレート}\\
\end{eqnarray*}
