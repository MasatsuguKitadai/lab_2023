\subsection*{【流れの解析解】}
\begin{eqnarray*}
  \zeta &=& z \sqrt{\frac{\omega}{\nu}}\\
  \\
  V_r &=& r \omega F \left(\zeta\right)\\
  V_\theta &=& r \omega G \left(\zeta\right)\\
  u &=& \sqrt{\nu \omega} H \left(\zeta\right)
\end{eqnarray*}

\subsection*{【境界条件】}
\begin{table}[hbtp]

  \begin{tabular}{ c c c c }

    $x=\infty$ & $V_r=0$ & $V_\theta =r \omega$ &       \\
  \end{tabular}
\end{table}
